LEFT JOIN

SELECT M.name
FROM MEAL M
LEFT JOIN INGREDIENT I
ON M.id = I.meal_id;

SELECT M.name
FROM MEAL M
LEFT JOIN INGREDIENT I
ON M.id = I.meal_id
WHERE M.id IS NULL;


RIGHT JOIN

SELECT M.name
FROM MEAL M
RIGHT JOIN INGREDIENT I
ON M.id = I.meal_id

SELECT M.name
FROM MEAL M
RIGHT JOIN INGREDIENT I
ON M.id = I.meal_id
WHERE I.meal_id IS NULL;


INNER JOIN

SELECT M.name
FROM MEAL M
INNER JOIN INGREDIENT I 
ON M.id = I.meal_id;
